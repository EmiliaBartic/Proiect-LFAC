%{
    #include<stdio.h>
    #include<string.h>
    #include<stdlib.h>
    #include<ctype.h>
    
    void yyerror(const char *s);
    int yylex();
    int yywrap();
    extern int nr_linie;
%}



%token CHARACTER PRINTFF SCANFF INTV FLOATV BOOLV CHARV CLASS STRUCT 
%token INT FLOAT BOOL CHAR VOIDF INTF BOOLF CHARF FLOATF FOR IF ELSE TRUE FALSE NUMBER FLOAT_NUM ID LE GE EQ NE GT LT 
%token AND OR STR ADD MULTIPLY DIVIDE SUBTRACT UNARY INCLUDE RETURN 
%%

program: headers function main '(' ')' '{' body return '}' {printf("program corect sintactic\n");}
;

headers: headers headers
| INCLUDE
;

main: type_function ID
;



type_function: VOIDF
| INTF
| BOOLF
| CHARF
| FLOATF
;

function: type_function ID '(' parametrii ')' '{' statement '}'
;

parametrii: datatype_var ID ',' datatype_var ID
| datatype_var ID
| datatype_vector ID '['value']' ',' datatype_vector ID '['value']'
| datatype_vector ID '['value']'
;

datatype_var: INT
| FLOAT 
| BOOL 
| CHAR 
;

datatype_vector : INTV
| FLOATV
| BOOLV
| CHARV
;

body: FOR '(' statement ';' condition ';' statement ')' '{' body '}'
| IF '(' condition ')' '{' body '}' else
| statement ';' 
| body body
| PRINTFF '(' STR ')' ';'
| SCANFF '(' STR ',' '&' ID ')' ';'
;

else: ELSE '{' body '}'
|
;

condition: value relop value 
| TRUE 
| FALSE
;

statement: datatype_var ID init 
| ID '=' expression 
| ID relop expression
| ID UNARY 
| UNARY ID
| datatype_vector ID '['value']'
| datatype_vector ID '['']' '=' '{' valori_vector '}'
| CLASS ID '{' clasa_struct_declaratii '}'
| STRUCT ID '{' clasa_struct_declaratii '}'
;

clasa_struct_declaratii : datatype_var ID ';'
| datatype_vector ID '['value']' ';'
;

valori_vector: value ',' value
| value
;
init: '=' value 
|
;

expression: expression arithmetic expression
| value
;

arithmetic: ADD 
| SUBTRACT 
| MULTIPLY
| DIVIDE
;

relop: LT
| GT
| LE
| GE
| EQ
| NE
;

value: NUMBER
| FLOAT_NUM
| CHARACTER
| ID
;

return: RETURN value ';' 
|
;

%%

int main() {
    yyparse();
}
